        "list_notes = \"\"\"\n",
        "import { db } from '@/lib/firebase';\n",
        "import {\n",
        "  collection,\n",
        "  query,\n",
        "  orderBy,\n",
        "  onSnapshot,\n",
        "} from 'firebase/firestore';\n",
        "\n",
        "export function watchNotes(uid: string, cb: (docs: any[]) => void) {\n",
        "  const q = query(\n",
        "    collection(db, `users/${uid}/notes`),\n",
        "    orderBy('updatedAt', 'desc')\n",
        "  );\n",
        "\n",
        "  return onSnapshot(q, (snap) =>\n",
        "    cb(\n",
        "      snap.docs.map((d) => ({\n",
        "        id: d.id,\n",
        "        ...d.data(),\n",
        "      }))\n",
        "    )\n",
        "  );\n",
        "}\n",

        "firestore_rules = \"\"\"\n",
        "rules_version = '2';\n",
        "\n",
        "service cloud.firestore {\n",
        "  match /databases/{database}/documents {\n",
        "\n",
        "    function isSignedIn() {\n",
        "      return request.auth != null;\n",
        "    }\n",
        "\n",
        "    function isOwner(uid) {\n",
        "      return isSignedIn() && request.auth.uid == uid;\n",
        "    }\n",
        "\n",
        "    match /users/{uid} {\n",
        "\n",
        "      allow read, write: if isOwner(uid);\n",
        "\n",
        "      match /notes/{noteId} {\n",
        "\n",
        "        allow create: if isOwner(uid);\n",
        "\n",
        "        allow read, update, delete:\n",
        "          if isOwner(uid)\n",
        "          ||\n",
        "          (\n",
        "            resource.data.shared == true\n",
        "            &&\n",
        "            request.time < resource.data.shareExpiresAt\n",
        "          );\n",
        "      }\n",
        "    }\n",
